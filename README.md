# Gradle Work

## Setup Gradle

curl -s get.gvmtool.net | bash

source "/Users/xxx/.gvm/bin/gvm-init.sh"

gvm install gradle


## Command

gradle tasks

gradle properties

gradle clean

gradle init --type java-library

gradle --gui


## Options

gradle -i(--info)

gradle -s(--stacktrace)

gradle -S(--full-stacktrace)

gradle -d(--debug)

gradle -q(--quiet)

gradle -b(--build-file) [gradle file]

gradle -?(-h or --help)
